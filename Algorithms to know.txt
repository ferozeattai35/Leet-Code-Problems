**Variable types to know**

NOTE: In C, The asterisk (*) is used to declare pointers. A pointer is a variable that doesn't
store a value like (5 or 3.14) but instead stores the memory address of another variable. 
Think of an int as a house and a pointer as the GPS coordinates to that house.

1.) Basic Types (The Big Four):
    1.) int: use for whole numbers (no decimals) like ages, counts, or loops.
    2.) float: Use for decimal numbers (fractions) where you don't need extreme precision, like a price or a temparture.
    3.) double: Use for precise decimals. It is double precision meaning it is more accurate
    4.) char: Use for single characters like 'A','b','$' in C, a char is actually a small integer that stores the character's ASCII code

2.) Modifiers (Fine tuning your choice):
    1.) Unsigned: Use when you know a number will never be negative
    2.) Long: Use for a very large whole number that might exceed 2 billion.
    3.) short: Use to save memory if you are storing thousands of small numbers.

3.) Quick Rule of Thumb:
    1.) Counting things? Use int.
    2.) Doing match with decimals? Default to double for better accuracy unless memory is tight.
    3.) Storing text? Use a char array (string) or a single char.
    4.) Only True/False? use bool (available in <stdbool.h)

1.)Searching and Sorting
    1.)Binary Search
        >The most efficient way to find an element in a sorted list by repeatedly halving the search range.
        >Coding: 
        int binarySearch(int arr[], int size, int target){
            int low=0,high = size - 1;
            while(low<=high){
                int mid = low+(high-low)/2;
                if(arr[mid] == target) return mid;
                if(arr[mid] < target) low = mid + 1;
                else high = mid -1;
            }
            return -1;
        }

2.) Graph and Tree algorithms
    1.)Breadth-First Search (BFS)
        >Explores a graph level-by-level; ideal for finding the shortest path in unweighted graphs.
        >Coding:
        void bfs(int adj[MAX][MAX], int startNode, int totalNodes){
            int visited[MAX] = {0}, queue[MAX], head = 0, tail = 0;
            visited[startNode] = 1;
            queue[tail++] = startNode;
            while(head < tail){
                int current = queue[head++]{
                    for(int i = 0; i < totalNodes; i++){
                        if(adj[current][i] && !visited[i]){
                            visited[i] = 1;
                            queue[tail++] = i;
                        }
                    }
                }
            }
        }
    2.)Depth-First Search (DFS)
        >Explores as far as possible along a branch before backtracking; useful for pathfinding and cycle detection
        >Most commonly implemented using recursion, which literally uses the program's stack to remember which node to visit next.
        >Coding: 
        void dfs(int graph[MAX_NODES][MAX_NODES], bool visited[MAX_NODES], int node){
            visited[node] = true;
            printf("Visited Node: %d\n", node);

            for(int i = 0; i < MAX_NODES; i++){
                if(graph[node][i] == 1 && !visited[i]){
                    dfs(graph, visited, i);
                }
            }
        }

        void dijkstra(int graph[V][V], int src){
            int dist[V];
            bool visited[V];

            for(int i = 0; i < V; i++){

        //Tree Traversal is typically done on a binary search tree (BST)
        //**Basic structure for our tree node**
        struct Node{
            int data;
            struct Node* left;
            struct Node* right;
        };
        //**In-order TRaversal (left, root, right)
        void inorder(struct Node* root){
            if(root != NULL){
                inorder(root -> left);
                printf("%d", root -> data);
                inorder(root->right);
            }
        }
        //**Pre-order Traversal(root, left, right)
        void preorder(struct Node* root){
            if(root != NULL){
                printf("%d", root->data);
                preorder(root -> left);
                preorder(root -> right);
            }
        }
        //**Post-order Traversal(left, right, root)
        void postorder(struct Node* root){
            if(root != NULL){
                postorder(root -> left);
                postorder(root -> right);
                printf("%d", root -> data);
            }
        }
        //**Full Example: Building and Printing
        struct Node* newNode(int data){
            struct Node* node = (struct Node*)malloc(sizeof(struct Node));
            node -> data = data;
            node -> left = node -> right = NULL;

3.) Algorithmic Techniques and Patterns.
    1.)Two Pointers
        >Moving two indices (e.x. from opposite ends) to find pairs or subarrays in a sorted list.
        >Coding: 
        void twoSumSorted(int arr[], int size, int target){
            int left = 0; right = size - 1;
            while(left < right){
                int sum = arr[left] + arr[right];
                if(sum == target){
                    printf("indices: %d, %d\n", left, right);
                    return;
                }
                if(sum<target) left++;
                else right--;
            }
        }
    2.)Sliding Window
        >Maintaining a sub-range of data that shifts across an array to solve problems like "Longest substring"
        >Coding:
        int maxSubarraySum(int arr[], int n, int k){
            for(int i = 0; i < k; i++){
                window_sum += arr[i];
                int max_sum = window_sum;
                for(int i = k; i < n; i++){
                    if(window_sum > max_sum) max_sum = window_sum;
                }
                return max_sum
            }
        }


    3.)Dynamic Programming
        >Breaking a complex problem into smaller subproblems and storing the results to avoid redundant work
        >Coding: 
        long long fibDP(int n){
            long long f[n+1];
            f[0] = 0; f[1] = 1;
            for(int i = 2; i <= n; i++){
                f[i] = f[i - 1] + f[i - 2];
            }
            return f[n];
        }
    4.)Greedy algorithms
        >Making the locally optimal choice at each step to find a global optimum
        >Coding:
        void findMinCoins(int amount){
            int coins[] = {25,10,5,1};
            int n = size(coins) / sizeof(coins[0]);

            printf("Coins for %d cents: \n", amount);

            for(int i = 0; i < n; i++){

        //**Brute Force
        void naiveSearch(char* pat, char* txt){
            int M = strlen(pat);
            int N = strlen(txt);

            for(int i = 0; i <= N - M; i++){
                int j;
                for(j = 0; j < M; j++){
                    if(txt[i + j] != pat[j])
                    break;
                }
                if(j == M){
                    printf("Pattern found at index %d\n", i);
                }
            }
        }

        //**(KMP) algorithm
        void computerLPS(char* pat, int M, int* lps){
            int len = 0, i = 1;
            lps[0] = 0;
            while(i < M){
                if(pat[i] == pat[len]){
                    lps[i++] = ++len;
                }else if(len != 0){
                    len = lps[len - 1];
                }else{
                    lps[i++] = 0;
                }
            }
        }
        void KPMSearch(char* pat, char* txt){
            int M = strlen(pat), N = strlen(txt);
            int *lps = (int*)malloc(sizeof(int) * M);
            computerLPS(pat, M, lps);

            int i = 0, j = 0;
            while((N - i) >= (M - j)){
                if(pat[j] == txt[i]){
                    i++,j++;
                }
                if(j == M){
                    printf("found patter at index %d\n", i - j);
                    j = lps[j - 1];
                }else if(i < N && pat[j] != txt[i]){
                    if(j != 0) j = lps[j - 1];
                    else i++;
                }
            }
            free(lps);
        }

        >Using binary operators (AND, OR, XOR) for highly optimized low-level operations
        >Coding: 
        bool isOdd(int n){
            return (n & 1);
        }

        int getBit(int num, int position){
            return (num >> position) & 1;
        }
        int setBit(int num, int position){
            return num | (1 << position);
        }

        int clearBit(int num, int position){
            return num & ~(1 << position);
        }

        bool isPowerofTwo(int n){
            return (n > 0) && ((n & (n-1))==0);
        }
